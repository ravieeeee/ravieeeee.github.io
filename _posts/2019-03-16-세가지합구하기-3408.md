# 문제

# 코드
```c++
	
#include <iostream>
 
using namespace std;
 
int main() {
    int tc;
    cin >> tc;
    long long n, s1, s2, s3;
    for (int i = 1; i <= tc; i++) {
        n = 0; s1 = 0; s2 = 0; s3 = 0;
        cin >> n;
        s1 = n * (n + 1) / 2;
 
        long long max_even_num = 2 * n;
        long long pair = 2 + max_even_num;
        s3 += pair * (n / 2);
        if (n % 2 != 0) s3 += 2 * ((n / 2) + 1);
        s2 = s3 - n;
 
        cout << "#" << i << " " << s1 << " " << s2 << " " << s3 << endl;
    }
    return 0;
}
```
