# 코드
```c++
#include "3456.h"
#include <iostream>

using namespace std;

void solve_3456() {
    int tc;
    cin >> tc;

    for (int i = 1; i <= tc; i++) {
        int arr[101] = { 0 };
        int a, b, c;
        cin >> a >> b >> c;
        arr[a]++; arr[b]++; arr[c]++;
        for (int j = 1; j <= 100; j++) {
            if (arr[j] != 0 && arr[j] % 2 == 1) {
                cout << "#" << i << " " << j << endl;
                break;
            }
        }
    }
}
```
