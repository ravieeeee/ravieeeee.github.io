# 코드
```c++
#include "4299.h"
#include <iostream>

using namespace std;

bool compare(int d, int h, int m) {
    return (11 > d) || (d == 11 && 11 > h) || (d == 11 && h == 11 && 11 > m);
}

int diff(int d, int h, int m) {
    int result = 0;
    if (m - 11 < 0) {
        h--;
        m += 60;
    }
    result += m - 11;

    if (h - 11 < 0) {
        d--;
        h += 24;
    }
    result += (h - 11) * 60;
    result += (d - 11) * 1440;

    return result;
}

void solve_4299() {
    int tc;
    cin >> tc;
    for (int i = 1; i <= tc; i++) {
        int d, h, m;
        cin >> d >> h >> m;

        if (compare(d, h, m)) {
            cout << "#" << i << " " << -1 << endl;
            continue;
        }

        cout << "#" << i << " " << diff(d, h, m) << endl;
    }
}
```
