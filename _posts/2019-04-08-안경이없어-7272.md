# 코드
```c++
#include "7272.h"
#include <iostream>

using namespace std;


void solve_7272() {
    int tc;
    cin >> tc;
    for (int i = 1; i <= tc; i++) {
        string a, b;
        cin >> a >> b;

        if (a.size() != b.size()) {
            cout << "#" << i << " DIFF" << endl;
            continue;
        }

        bool flag = true;
        for (int j = 0; j < a.size(); j++) {
            if (a.at(j) == 'B') {
                if (b.at(j) == 'B') continue;
                else {
                    cout << "#" << i << " DIFF" << endl;
                    flag = false;
                    break;
                }
            }
            if (a.at(j) == 'A' || a.at(j) == 'D' || a.at(j) == 'O' || a.at(j) == 'P' ||
                    a.at(j) == 'Q' || a.at(j) == 'R') {
                if (b.at(j) == 'A' || b.at(j) == 'D' || b.at(j) == 'O' || b.at(j) == 'P' ||
                    b.at(j) == 'Q' || b.at(j) == 'R') continue;
                else {
                    cout << "#" << i << " DIFF" << endl;
                    flag = false;
                    break;
                }
            }
            if (b.at(j) == 'A' || b.at(j) == 'D' || b.at(j) == 'O' || b.at(j) == 'P' ||
                b.at(j) == 'Q' || b.at(j) == 'R' || b.at(j) == 'B') {
                cout << "#" << i << " DIFF" << endl;
                flag = false;
                break;
            }
        }

        if (flag) cout << "#" << i << " SAME" << endl;
    }
}
```
